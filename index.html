// MPPA — Apps Script v5 // Novidade: upload de PDF para Google Drive const SHEET_CONFIG = 'config'; const SHEET_CODES = 'codigos'; const DRIVE_FOLDER = 'MPPA — Minutas'; // Nome da pasta no Drive (criada automaticamente) // ── Helpers ────────────────────────────────────────────────────── function getSheet(name) { const ss = SpreadsheetApp.getActiveSpreadsheet(); let sheet = ss.getSheetByName(name); if (!sheet) { sheet = ss.insertSheet(name); if (name === SHEET_CODES) { sheet.getRange('A1').setValue('Código'); sheet.getRange('B1').setValue('URL'); sheet.getRange('C1').setValue('Salvo em'); } if (name === SHEET_CONFIG) { sheet.getRange('A1').setValue('Chave'); sheet.getRange('B1').setValue('Valor'); } } return sheet; } function getDataSheet(tab) { const name = tab || 'dados'; const ss = SpreadsheetApp.getActiveSpreadsheet(); let sheet = ss.getSheetByName(name); if (!sheet) { sheet = ss.insertSheet(name); sheet.getRange('A1').setValue('Última atualização'); sheet.getRange('B1').setValue('JSON'); } return sheet; } function getDriveFolder() { const folders = DriveApp.getFoldersByName(DRIVE_FOLDER); if (folders.hasNext()) return folders.next(); return DriveApp.createFolder(DRIVE_FOLDER); } // ── GET ────────────────────────────────────────────────────────── function doGet(e) { const action = e && e.parameter && e.parameter.action; const tab = e && e.parameter && e.parameter.tab; if (action === 'load') { try { const sheet = getDataSheet(tab); const lastRow = sheet.getLastRow(); if (lastRow < 2) return jsonResponse({ ok: true, data: null }); const json = sheet.getRange('B' + lastRow).getValue(); return jsonResponse({ ok: true, data: JSON.parse(json) }); } catch (err) { return jsonResponse({ ok: false, error: err.message }); } } if (action === 'getCode') { try { const code = (e.parameter.code || '').toLowerCase(); if (!code) return jsonResponse({ ok: false, error: 'Código vazio' }); const sheet = getSheet(SHEET_CODES); const lastRow = sheet.getLastRow(); if (lastRow < 2) return jsonResponse({ ok: false, error: 'Nenhum código' }); const data = sheet.getRange(2, 1, lastRow - 1, 2).getValues(); for (const [c, url] of data) { if (c === code && url) return jsonResponse({ ok: true, url }); } return jsonResponse({ ok: false, error: 'Código não encontrado' }); } catch (err) { return jsonResponse({ ok: false, error: err.message }); } } if (action === 'getConfig') { try { const sheet = getSheet(SHEET_CONFIG); const lastRow = sheet.getLastRow(); if (lastRow < 2) return jsonResponse({ ok: true, pwdHash: null }); const data = sheet.getRange(2, 1, lastRow - 1, 2).getValues(); const config = {}; for (const [key, val] of data) { if (key) config[key] = val; } return jsonResponse({ ok: true, pwdHash: config['pwdHash'] || null }); } catch (err) { return jsonResponse({ ok: false, error: err.message }); } } return jsonResponse({ ok: false, error: 'Ação desconhecida: ' + action }); } // ── POST ───────────────────────────────────────────────────────── function doPost(e) { try { const body = JSON.parse(e.postData.contents); const tab = body.tab; // ── save: salva JSON dos processos ─────────────────────────── if (body.action === 'save') { const sheet = getDataSheet(tab); const now = Utilities.formatDate(new Date(), 'America/Belem', 'dd/MM/yyyy, HH:mm:ss'); const json = JSON.stringify(body.data); const lastRow = Math.max(sheet.getLastRow(), 1); sheet.getRange(lastRow + 1, 1).setValue(now); sheet.getRange(lastRow + 1, 2).setValue(json); if (lastRow > 31) sheet.deleteRow(2); return jsonResponse({ ok: true, savedAt: now }); } // ── uploadFile: recebe PDF em base64, salva no Drive ───────── if (body.action === 'uploadFile') { const fileName = body.fileName || ('minuta_' + Date.now() + '.pdf'); const base64 = body.base64; const mimeType = body.mimeType || 'application/pdf'; const procNum = body.procNum || ''; if (!base64) return jsonResponse({ ok: false, error: 'Arquivo vazio' }); // Limit: ~8MB base64 ≈ 6MB file if (base64.length > 11000000) { return jsonResponse({ ok: false, error: 'Arquivo muito grande (máx 6MB)' }); } const blob = Utilities.newBlob( Utilities.base64Decode(base64), mimeType, fileName ); const folder = getDriveFolder(); // Create subfolder per process if number provided let targetFolder = folder; if (procNum) { const sub = folder.getFoldersByName(procNum); targetFolder = sub.hasNext() ? sub.next() : folder.createFolder(procNum); } const file = targetFolder.createFile(blob); file.setSharing(DriveApp.Access.ANYONE_WITH_LINK, DriveApp.Permission.VIEW); const url = file.getUrl(); return jsonResponse({ ok: true, url, fileId: file.getId(), fileName }); } // ── saveCode ───────────────────────────────────────────────── if (body.action === 'saveCode') { const code = (body.code || '').toLowerCase().replace(/[^a-z0-9]/g, ''); const url = body.url || ''; if (!code || !url) return jsonResponse({ ok: false, error: 'Dados incompletos' }); const sheet = getSheet(SHEET_CODES); const now = Utilities.formatDate(new Date(), 'America/Belem', 'dd/MM/yyyy HH:mm'); const lastRow = sheet.getLastRow(); if (lastRow >= 2) { const codes = sheet.getRange(2, 1, lastRow - 1, 1).getValues().flat(); const rowIdx = codes.indexOf(code); if (rowIdx >= 0) { sheet.getRange(rowIdx + 2, 2).setValue(url); sheet.getRange(rowIdx + 2, 3).setValue(now); return jsonResponse({ ok: true, updated: true }); } } sheet.appendRow([code, url, now]); return jsonResponse({ ok: true, created: true }); } // ── saveConfig ─────────────────────────────────────────────── if (body.action === 'saveConfig') { const sheet = getSheet(SHEET_CONFIG); const now = Utilities.formatDate(new Date(), 'America/Belem', 'dd/MM/yyyy HH:mm'); const updates = {}; if (body.pwdHash) updates['pwdHash'] = body.pwdHash; const lastRow = sheet.getLastRow(); for (const [key, val] of Object.entries(updates)) { let found = false; if (lastRow >= 2) { const keys = sheet.getRange(2, 1, lastRow - 1, 1).getValues().flat(); const rowIdx = keys.indexOf(key); if (rowIdx >= 0) { sheet.getRange(rowIdx + 2, 2).setValue(val); found = true; } } if (!found) sheet.appendRow([key, val, now]); } return jsonResponse({ ok: true }); } return jsonResponse({ ok: false, error: 'Ação desconhecida.' }); } catch (err) { return jsonResponse({ ok: false, error: err.message }); } } function jsonResponse(obj) { return ContentService .createTextOutput(JSON.stringify(obj)) .setMimeType(ContentService.MimeType.JSON); }
